let digito (c:char) = 
match c with
    '1'|'2'|'3'|'4'|'5'|'6'|'7'|'8'|'9'|'0' -> true
|_ -> false;;

(*Quest達o 2*)

let minuscula (l:char) = 
match l with
 'A' .. 'Z' -> Char.lowercase l
| _ -> l;;

let avaliador ((op1,operador,op2):int*char*int) = 
match operador with
 '+'-> op1+op2
|'-'-> op1-op2
|'*'-> op1*op2
|'/'-> op1/op2
|_ -> -1;;
